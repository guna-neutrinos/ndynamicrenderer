import { Inject, Injectable } from '@angular/core';
import { Validators } from '@angular/forms';
import * as _ from 'lodash';
import Ajv from 'ajv';
import addFormats from 'ajv-formats';
import { Config, Metadata } from './interfaces/interface';

@Injectable({
  providedIn: 'root',
})
export class NDynamicRendererService {
  modelSchema: any;
  ajv: any;
  rendererConfig: Config[] = [];
  constructor(@Inject('config') private _config: any) {
    this.rendererConfig = _config || [];
    this.ajv = new Ajv();
    addFormats(this.ajv);
    this.modelSchema = {};
  }

  componentCache: any = {};

  getComponents(metadata: Metadata) {
    if (this.componentCache[metadata.component_name]) {
      return this.componentCache[metadata.component_name];
    } else {
      return this.getNewComponent(metadata);
    }
  }

  public get config(): Config[] {
    return this.rendererConfig;
  }

  public set resetConfig(config: Config[]) {
    this.rendererConfig = config;
  }
  /**
   * This function decides which component to render.
   * @param metadata {Metadata} - Metadata which contains the component information.
   * @returns Component Instance
   */

  getNewComponent(metadata: Metadata) {
    let template = '';
    const findLibrary = (source: any, name: string) =>
      source.find((el: any) => el.library === name);
    const library = findLibrary(this.rendererConfig, metadata.library);
    if (library?.imports && typeof library.imports == 'object') {
      template = library.imports[metadata.component_name];
    }
    this.componentCache[metadata.component_name] = template;
    return template;
  }

  /**
   * Reads the model  and returns the value
   * @param model_path  {string}  - model_path
   * @returns {any}- Value of the model_path from the model object.
   */
  readModel(model_path: string, model: object) {
    return _.get(model, model_path) || '';
  }

  /**
   * Gets Validators of an individual property from the schema.
   * @param schema {object} - The model schema.
   * @param model_path {string} - The model_path for which to get the validators from the schema.
   * @returns {object} - The validators from the schema.
   */

  getValidatorsFromSchema(schema: any, model_path: string) {
    let schemaPath = this.getPathForSchema(model_path);
    let validators = _.get(schema, schemaPath);
    return validators;
  }

  /**
   * Prepares path to traverse through the schema from model_path .
   * @param model_path {string} - For which to create a path.
   * @returns {string} - Path to get the validators from.
   */

  getPathForSchema(model_path: any) {
    if (model_path) {
      model_path = model_path.split('.') || [];
      let path = '';
      for (let i = 0; i < model_path.length; i++) {
        if (model_path[i].includes('[')) {
          let str = model_path[i].split('[');
          str[1] = 'items.';
          model_path[i] = str.join('.');
          path = path + model_path[i];
          if (i == 0) path = 'items.' + path + model_path[i];
        } else {
          path =
            i == 0
              ? 'properties.' + path + model_path[i] + '.properties.'
              : i == model_path.length - 1
              ? path + model_path[i]
              : path + model_path[i] + '.properties.';
        }
      }
      return path;
    }
    return '';
  }
  /**
   *Prepares Validators to the formcontrol
   * @param validators {object} - An object of validators from the schema
   * @returns {Array} - An array of Formcontrol Validators
   */

  prepareValidators(validators: object) {
    if (!validators) return [];
    const validatorsToAdd = [];
    for (const [key, value] of Object.entries(validators)) {
      switch (key) {
        case 'min':
          validatorsToAdd.push(Validators.min(value as number));
          break;
        case 'max':
          validatorsToAdd.push(Validators.max(value as number));
          break;
        case 'required':
          if (value) {
            validatorsToAdd.push(Validators.required);
          }
          break;
        case 'requiredTrue':
          if (value) {
            validatorsToAdd.push(Validators['required']);
          }
          break;
        case 'format':
          if (value == 'email') {
            validatorsToAdd.push(Validators.email);
          }
          break;
        case 'minLength':
          validatorsToAdd.push(Validators.minLength(value as number));
          break;
        case 'maxLength':
          validatorsToAdd.push(Validators.maxLength(value as number));
          break;
        case 'pattern':
          validatorsToAdd.push(Validators.pattern(value as string));
          break;
        case 'nullValidator':
          if (value) {
            validatorsToAdd.push(Validators.nullValidator);
          }
          break;
        default:
          break;
      }
    }
    return validatorsToAdd || [];
  }

  /**
   * Gets the value of the path from the model
   * @param model_path {string}- The model path
   * @param model {object} - The object from which to get the value from.
   * @returns {any}- The value from the model object.
   */

  readValue(model_path: string, model: object) {
    if (typeof model_path == 'string' && model_path?.startsWith('@')) {
      return this.readModel(model_path.substr(1), model);
    }
    return model_path;
  }

  /**
   * Evaluates the expression provided.
   * @param expr - Expression that needs to be evaluated.
   * @returns {Boolean}- Evaluated result.
   */
  evaluateExpression(expr: any, model: any) {
    const key = Object.keys(expr)[0];
    switch (key) {
      case '$or':
        return expr[key].some((c: any) =>
          typeof c === 'object'
            ? this.evaluateExpression(c, model)
            : this.readValue(c, model)
        );
      case '$and':
        return expr[key].every((c: any) =>
          typeof c === 'object'
            ? this.evaluateExpression(c, model)
            : this.readValue(c, model)
        );
      case '$gt':
        return (
          this.readValue(expr[key][0], model) >
          this.readValue(expr[key][1], model)
        );
      case '$lt':
        return (
          this.readValue(expr[key][0], model) <
          this.readValue(expr[key][1], model)
        );
      case '$eq':
        return (
          this.readValue(expr[key][0], model) ==
          this.readValue(expr[key][1], model)
        );
      case '$ne':
        return (
          this.readValue(expr[key][0], model) !==
          this.readValue(expr[key][1], model)
        );

      default:
        throw 'invalid expression';
    }
  }

  /**
   * Validates the object with the schema.
   * @param obj {object} - The object to be validated.
   * @param schema {object} - Schema.
   * @returns {Boolean} - Returns a boolean based on the validation.
   */

  validateObjectWithSchema(obj: any, schema: any) {
    return this.ajv.validate(schema, obj);
  }

  /**
   * This function helps to deepclone arrays.
   */
  load_deep(a: any, b: any) {
    if (_.isArray(a) && _.isArray(b)) {
      return a.concat(b);
    }
    return;
  }

  /**
   * This deepClones the objects.
   * @param model - The object with which you want to deep merge.
   * @param form - The object which you want to deep merge.
   * @returns {object} - The deepmerged object.
   */
  mergeDeep(model: any, form: any) {
    return _.mergeWith(model, form, this.load_deep);
  }

  /**
   * Creates an unflatten Object with the given key value pair
   * @param data  - The object with key value pair
   * @returns {Object} -Unflattened Object
   */

  createUnflattenObject(data: any) {
    if (Object(data) !== data || Array.isArray(data)) return data;
    const regex = /\.?([^.\[\]]+)$|\[(\d+)\]$/;
    const props = Object.keys(data);
    let result, p;
    while ((p = props.shift())) {
      const m: any = regex.exec(p);
      let target;
      if (m.index) {
        const rest = p.slice(0, m.index);
        if (!(rest in data)) {
          data[rest] = m[2] ? [] : {};
          props.push(rest);
        }
        target = data[rest];
      } else {
        if (!result) {
          result = m[2] ? [] : {};
        }
        target = result;
      }
      target[m[2] || m[1]] = data[p];
    }
    return result;
  }

  /**
   *The modelPath has an '@' decorator which needs to removed to compute it's value
   * @param str - The model path with @decorator
   * @returns {string} - The actual model_path
   */
  trimModelValue(str: string) {
    if (str?.startsWith('@')) {
      return str.substr(1);
    }
    return str;
  }

  /**
   * Updates the Model
   * @param formgroupVal  - The formGroup value which contains the changes
   * @param model - The model
   * @param schema - The schema
   * @param modelChange - The callback which sends the updated model
   * @param errors - The callback which sends errors if there are any.
   */

  updateObject(
    formgroupVal: any,
    model: any,
    schema: any,
    modelChange: any,
    errors: any
  ) {
    const unflatObj = this.createUnflattenObject(formgroupVal);
    const valid = this.validateObjectWithSchema(unflatObj, schema);

    if (valid) {
      let deepData = this.mergeDeep(model, unflatObj);
      modelChange(deepData);
    }
    if (!valid) {
      errors(this.ajv?.errors);
    }
  }

  /**
   * Validate the return result from the global functions and updates the Model
   * @param result {object} - The return value from the global functions.
   * @param model - The model.
   * @param schema - The schema.
   * @param modelChange - The callback which sends the updated model
   * @param errors - The callback which sends errors if there are any.
   */
  validateResult(
    result: any,
    model: any,
    schema: any,
    modelChange: any,
    errors: any
  ) {
    this.updateObject(result?.model, model, schema, modelChange, errors);
  }

  /**
   *Calls the global functions that got loaded on the intitialization of the application.
   * @param viewFunctions - The functions from the view which needs to be called.
   * @param model - The model.
   * @param view -The view.
   * @param schema - The Schema
   * @param modelChange - The callback which sends the updated model
   * @param errors - The callback which sends errors if there are any.
   */
  async callFunction(
    viewFunctions: any,
    model: object,
    view: any,
    schema: any,
    modelChange: any,
    errors: any
  ) {
    this.modelSchema = schema;
    const globalMethods: any = window['NRenderer' as keyof Window];

    for (let i of viewFunctions) {
      if (globalMethods[i?.functionName]) {
        if (i?.arguments) {
          await globalMethods[i?.functionName].apply({}, [
            view,
            schema,
            (result: any) => {
              this.validateResult(result, model, schema, modelChange, errors);
            },
            ...i?.arguments,
          ]);
        } else {
          await globalMethods[i?.functionName].apply({}, [
            view,
            schema,
            (result: any) => {
              this.validateResult(result, model, schema, modelChange, errors);
            },
          ]);
        }
      }
    }
  }

  /**
   * Returns the type that can be passed to <input/>
   * @param type  - The type from the schema
   * @returns {string} -  Returns the type that can be passed to <input/>
   */

  getInputType(type: string) {
    switch (type) {
      case 'text':
        type = 'text';
        break;
      case 'number':
        type = 'number';
        break;
      case 'string':
        type = 'text';
        break;
      case 'boolean':
        type = 'boolean';
        break;
      default:
    }
    return type;
  }
}

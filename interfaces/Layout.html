export type Panel = Panels[];
/**
 * Each item in the Expansion Panel is containerised by a card and styles,layout can be set on the panel level.
 * ```text
 * panel_name - Name of the Expansion Panel Header.
 * expanded - Expansional Panel to be expanded or not.
 * styles- A key value pair of the CSS Styles.
 * class - string | string[]
 * layout - Responsive flex properties for mobile,tablet,desktop and tv
 * ```
 */
export interface Panels {
  panel_name: string;
  expanded?: boolean;
  styles?: KeyValue;
  class?: string | string[];
  layout?: KeyValue;
  sections: Sections[];
}

export interface KeyValue {
  [key: string]: any;
}

export interface Sections {
  section: Section;
}

/**
 * Each item in the Section is containerised by a card and styles,layout can be set on the Section level.
 */
export interface Section {
  styles?: KeyValue;
  class?: string;
  layout?: KeyValue;
  leafs: Leaf[];
}

/**
 *  styles,layout can be set on the Leaf level too.
 */

export interface Leaf {
  styles?: KeyValue;
  class?: string;
  layout?: KeyValue;
  id?: string | number;
  metadata: Metadata;
}

type updateOn = 'change' | 'blur' | 'submit';

/**
 * ```
 * version- Version of the library.
 * library- Name of the library.
 * component_name - Name of the component that needs to be renderered.
 * options- The options are the inputs to the component.
 * expression - The condition that sets to show the component.
 * updateOn - The update strategy on the formControl
 * ```
 */
export interface Metadata {
  version: string;
  library: string;
  component_name: string;
  options: Options;
  expression?: KeyValue;
  updateOn: updateOn;
}
/**
 * 
 * Options are the inputs to the component.
 * ```
 * model_path - The model path with @ .Eg: @case.personaldetails.name
 * output_events - It is an object of functions that has to be triggered from the components during a change event
 * initialize_functions - It is an object of functions that has to be triggered from the components during initialization
 * formController - This is generated by the library and is passed to the component.
 * label - Label of the component
 * options - It is an array of options to the dropdown component
 * disabled -Boolean to determine whether the component is disabled.
 * readonly - Boolean to determine whether the component is readonly.
 * type - Type is pulled from schema and is used to define the type on formcontroller and to parse the value
 * [key: string] - It takes someOther inputs based on the user configuration
 * validations -  To ensure that the data entered by users is accurate and meets our application's requirements.
 * ```
 */
export interface Options {
  model_path?: string;
  output_events?: Function[];
  initialize_functions?:Function[];
  label?: KeyValue[];
  options?: any;
  formController?: any;
  type?: any;
  disabled?:boolean;
  readonly?:boolean;
  [key: string]: any;
  validations?:KeyValue;

}

/**
 * 
 * Validations are to ensure that the data entered by users is accurate and meets our application's requirements.
 * we have default angular validations as well as custom validations.
 * ```
 * type - the data type of the component.
 * dependencyValidations - Defines the validations that needs to be added on other components based on certain conditions of the current compoent.
 * Example - 
 "validations": {
                "type": "string",
                "required": true,
                "dependencyValidations": []
 }
 * ```
 */

export interface Validations{
  type?:string;
  required?:boolean;
  dependencyValidations?:DependencyValidations[] 
}

/**
 * Defines the validations that needs to be added on other components based on certain conditions of the current compoent.
 * validations - To ensure that the data entered by users is accurate and meets our application's requirements.
 * ```
 * required_value - This is a query which follows mongoDB query syntax which results a boolean.
 * current_model_path - Model path of the current component.
 * Example -
    {
    "validations": {
      "required": true,
      "idNoLengthCheckValidator": true
    },
    "required_value": {
      "$or": [
        {
          "$eq": ["@client_details.idType", "1"]
        },
        {
          "$eq": ["@client_details.idType", "7"]
        }
      ]
    },
    "target_model_path": "@client_details.idNo",
    "current_model_path": "@client_details.idType"
  }
 * 
 * ```
 * 
 */
export interface DependencyValidations{
  validations:KeyValue;
  required_value:KeyValue;
  target_model_path:KeyValue;
  current_model_path:KeyValue
}

/**
 * functionName - Name of the function.
 * Arguments - Which are optional are passed as arguments.
 */
export interface Function {
  functionName: string;
  arguments?: number[] | string[];
}

export interface Config {
  library: string;
  imports: any;
}
/**
 * Define flex for different screen sizes
 */
export interface Layout {
  mobile: Flex;
  tablet: Flex;
  desktop: Flex;
  tv: Flex;
}

export interface Flex {
  flex: string;
}

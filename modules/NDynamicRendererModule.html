import { ModuleWithProviders, NgModule } from '@angular/core';
import { NDynamicRendererComponent } from './n-dynamic-renderer/n-dynamic-renderer.component';
import { DynamicModule } from 'ng-dynamic-component';
import { DynamicIoModule  } from 'ng-dynamic-component';
import { MatExpansionModule } from '@angular/material/expansion';
import { MatInputModule } from '@angular/material/input';
import { FormsModule, ReactiveFormsModule } from '@angular/forms';
import { CommonModule } from '@angular/common';
import { FlexLayoutModule } from '@angular/flex-layout';
import { NDynamicRendererService } from './n-dynamic-renderer.service';
import { Config } from './interfaces/interface';
import { OverlayModule } from '@angular/cdk/overlay';
import {MatDialogModule} from '@angular/material/dialog';
import { MatSelectModule } from '@angular/material/select';

@NgModule({
  declarations: [NDynamicRendererComponent],
  imports: [
    MatExpansionModule,
    DynamicModule,
    MatInputModule,
    FormsModule,
    ReactiveFormsModule,
    CommonModule,
    FlexLayoutModule,
    OverlayModule,
    MatDialogModule,
    MatSelectModule ,
    DynamicIoModule
  ],
  exports: [NDynamicRendererComponent],
})
export class NDynamicRendererModule {
  constructor(private NDynamicRendererService: NDynamicRendererService) {}
  /**
   * Creates and configures a module with all the required libraries and their components provided.
   * @param config {Config[]}- The config which contains the libraryName and its imports.
   */
  static forRoot(
    config?: Config[]
  ): ModuleWithProviders<NDynamicRendererModule> {
    return {
      ngModule: NDynamicRendererModule,
      providers: [
        NDynamicRendererService,
        { provide: 'config', useValue: config },
      ],
    };
  }

  public get config() {
    return this.NDynamicRendererService.config;
  }
  /**
   * Resets the config i.e., the imports modules.
   * @param config
   */
  public resetConfig(config: Config[]) {
    this.NDynamicRendererService.resetConfig = config;
  }
}

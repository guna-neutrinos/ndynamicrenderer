import {
  Component,
  ComponentRef,
  EventEmitter,
  Input,
  OnInit,
  Output,
  SimpleChanges,
} from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
} from '@angular/forms';
import { NDynamicRendererService } from '../n-dynamic-renderer.service';
import Ajv from 'ajv';
import addFormats from 'ajv-formats';
import {
  ChangeEvent,
  Leaf,
  Metadata,
  Panel,
  Section,
  Sections,
} from '../interfaces/interface';

@Component({
  selector: 'n-dynamic-renderer',
  templateUrl: './n-dynamic-renderer.component.html',
  styleUrls: ['./n-dynamic-renderer.component.css'],
})
export class NDynamicRendererComponent implements OnInit {
  ajv: any;
  constructor(
    private nDynamicRendererService: NDynamicRendererService,
    private fb: FormBuilder
  ) {
    this.ajv = new Ajv();
    addFormats(this.ajv);
  }
  @Input() view: any;
  @Input() model: any;
  @Input() schema: any;
  @Input() appMetadata: any;
  @Output() modelChange = new EventEmitter();
  @Output() onError = new EventEmitter();
  @Output() changeDetection = new EventEmitter();
  formControlMapper: any = {};
  rendererId: any = '';
  viewPanels: any;
  flexVal: number = 100;
  public modelForm: any = {};
  typeMapper: any = {};
  compRef = {};
  outputs = {
    change_event: async (result: any) => {
      await this.outputMethods(result);
    },
    errors: async (error: any) => {
      this.errorEmit(error);
    },
  };

  componentCreated(compRef: ComponentRef<any>, metadata) {

    this.nDynamicRendererService.ComponentReference[
      metadata.options.model_path
    ] = compRef;
  }

  ngOnInit(): void {
    this.nDynamicRendererService.viewStore[this.rendererId] = this.view
    this.viewPanels = this.nDynamicRendererService.viewStore[this.rendererId]|| [];
    this.model = this.model || {};
    this.schema = this.schema || {};
    this.rendererId = Math.random().toString(16).slice(2);
    this.modelForm = this.nDynamicRendererService.modelForm;
    this.formControlMapper = this.nDynamicRendererService.formControlMapper;
    this.modelForm[this.rendererId] = this.fb.group({});
    this.formControlMapper[this.rendererId] = {};
    this.initView(this.viewPanels);
    this.formChangeDetection();
  }

  ngOnChanges(changes) {
    if (changes?.appMetadata?.currentValue?.cancel) {
      this.modelForm = {};
      this.formControlMapper = {};
      this.modelForm[this.rendererId] = this.fb.group({});
      this.formControlMapper[this.rendererId] = {};
      this.initView(this.viewPanels);
    }
    if (changes?.appMetadata?.currentValue?.expansions) {
      this.changeExpansionOnPanel(this.appMetadata.expansions);
    }
  }

  changeExpansionOnPanel(expansions) {
      this.view?.forEach((panel) => {
        if (panel && Object.keys(panel).length > 0) {
          panel['expanded'] = this.getExpansionFlag(panel, expansions);
        }
        panel?.sections.map((section) => {
          if (
            section &&
            Object.keys(section).length > 0 &&
            section.section &&
            Object.keys(section.section).length > 0
          ) {
            section.section['expanded'] = expansions[panel.panel_name];
          }
        });
      });
  
  }
  getExpansionFlag(panel, expansions) {
    if (panel && expansions) {
      if (expansions[panel.panel_name]) {
        return true;
      }
    }
    return false;
  }

  /**
   * Iterates through the view to create formControls
   * @param viewPanels - The view Object
   */
  initView(viewPanels: Panel) {
    try {
      for (let i = 0; i < viewPanels?.length; i++) {
        const panel = viewPanels[i];
        for (let j = 0; j < panel?.sections.length; j++) {
          const section = panel?.sections[j]?.section;
          for (let k = 0; k < section?.leafs?.length; k++) {
            let model_path = section.leafs[k]?.metadata?.options?.model_path;
            if (model_path) this.prepareFormControl(model_path, section, k);
          }
        }
      }
    } catch (error) {
      this.errorEmit(error);
    }
  }
  /**
   * Prepares FormControls
   * @param model_path - The Modelpath of the Model
   * @param section - The section key in the model
   * @param index - Index of the leaf in the section.
   */

  prepareFormControl(model_path: string, section: Section, index: number) {
    const id = Math.floor(Math.random() * 10000000);
    section.leafs[index].id = id;
    model_path = this.nDynamicRendererService.trimModelValue(model_path);
    const obj: any = {};
    obj[model_path] = this.nDynamicRendererService.readModel(
      model_path,
      this.model
    );
    const unflatObj = this.nDynamicRendererService.createUnflattenObject(obj);
    const valid = this.validateObjectWithSchema(
      unflatObj,
      this.schema,
      this.appMetadata
    );

    // if ( !valid || valid) {   //Commented for validation
    const value = this.nDynamicRendererService.readModel(
      model_path,
      this.model
    );

    if (this.modelForm[this.rendererId].value[model_path]) {
      this.formControlMapper[this.rendererId][id] =
        this.modelForm[this.rendererId].controls[model_path];
    } else {
      this.formControlMapper[this.rendererId][id] = new FormControl(value, {
        validators: this.nDynamicRendererService.prepareValidators(
          section.leafs[index].metadata?.options?.validations
        ),
        updateOn: section.leafs[index].metadata?.updateOn || 'blur',
      });
    }

    this.modelForm[this.rendererId].addControl(
      model_path,
      this.formControlMapper[this.rendererId][id]
    );

    this.typeMapper[model_path] =
      section.leafs[index].metadata?.options?.validations?.type || 'text';
    // }

    if (!valid) {
      this.ajv?.errors?.forEach((error: any) => this.errorEmit(error));
    }
  }

  /**
   * Returns the component instance to the Dynamic Renderer
   * @param metadata - Contains the information what component to render
   * @returns - Component Instance
   */
  getComponent(metadata: Metadata): any {
    try {
      return this.nDynamicRendererService.getComponents(metadata);
    } catch (error) {
      this.errorEmit(error);
    }
  }

  /**
   *Passes the inputs that needs to be passed as inputs to the Dynamic Renderer
   * @param leaf - Metadata which has inputs
   * @returns - Returns the inputs that needs to be passed as inputs to the Dynamic Renderer
   */
  getInputs(leaf: any) {
    try {
      if (leaf && leaf?.metadata && leaf?.id) {
        leaf.metadata.options.formController =
          this.formControlMapper[this.rendererId][leaf.id] ||
          new FormControl('');
        if (leaf.metadata.options?.model_path) {
          leaf.metadata.options.type =
            this.nDynamicRendererService.getInputType(
              leaf.metadata.options?.validations?.type || 'text'
            );
        }
        if (this.nDynamicRendererService.modifyOptions[leaf.id]) {
          leaf.metadata.options = {
            ...leaf.metadata.options,
            ...this.nDynamicRendererService.modifyOptions[leaf.id],
          };
        }

        leaf.metadata.options.edit = this.appMetadata?.edit;
        leaf.metadata.options.appMetadata = { ...this.appMetadata };
        leaf.metadata.options.componentRefs =
          this.nDynamicRendererService.ComponentReference;

        return leaf.metadata.options;
      }
      return leaf?.metadata?.options || {};
    } catch (error) {
      this.errorEmit(error);
    }
  }

  /**
   * Detects the formcontrol changes of all the formcontrols registered with the formGroup
   */

  formChangeDetection() {
    try {
      this.modelForm[this.rendererId].valueChanges.subscribe((v: any) => {
        // console.log(v);
        this.parseModel();
        this.detectErrorsOnFormControls();
        this.nDynamicRendererService.updateObject(
          this.modelForm[this.rendererId].value,
          this.model,
          this.schema,
          (updatedModel: any) => {
            this.modelChange.emit(updatedModel);
          },
          (errors: any) => {
            errors?.forEach((error: any) => this.errorEmit(error));
          }
        );
      });
    } catch (error) {
      this.errorEmit(error);
    }
  }
  touchedControls(modelForm: any) {
    let formControls = modelForm.controls;
    let touchedControls = {};
    for (let control in formControls) {
      if (formControls[control].touched) {
        touchedControls[control] = formControls[control];
      }
    }

    return touchedControls;
  }

  detectErrorsOnFormControls() {
    let message = '';
    let errors = this.collectErrors(this.modelForm[this.rendererId]);
    let touchedControls = this.touchedControls(this.modelForm[this.rendererId]);
    let disableObj = {};
    let rendererId = {};
    if (errors && Object.keys(errors)?.length > 0) {
      this.errorEmit(errors);
      rendererId[this.rendererId] = true;
    } else {
      this.errorEmit(errors);
      rendererId[this.rendererId] = false;
    }
    disableObj['disableSave'] = rendererId;
    this.changeDetection.emit(disableObj);
    let errorMessages = {};
    if (errors) {
      for (let error in errors) {
        let field = error
          .split('.')
          .pop()
          .replace(/([a-z0-9])([A-Z])/g, '$1 $2')
          .replace('-', '')
          .replace('_', ' ')
          .replace('-', ' ')
          .replace('_', ' ');
        field = field.charAt(0).toUpperCase() + field.slice(1);

        if (error) {
          message = `${field} is invalid`;
          if (errors[error]['required']) {
            message = `${field} is a required property.`;
          }
          if (errors[error]['minlength']) {
            message = ` The minimum length for the field ${field} is ${errors[error].minlength.requiredLength} characters.`;
          }
          if (errors[error]['maxlength']) {
            message = ` The maximum length for the field ${field} is ${errors[error].maxlength.requiredLength} characters.`;
          }
          if (errors[error]['min']) {
            message = ` The minimum value for the field ${field} is ${errors[error].min.min}.`;
          }
          if (errors[error]['max']) {
            message = ` The maximum value for the field ${field} is ${errors[error].max.max}.`;
          }
          if (errors[error]['email']) {
            message = `Enter a valid email address.`;
          }
          if (errors[error]['pattern']) {
            message = `The entered value doesn't match the pattern ${errors[error].pattern.requiredPattern}.`;
          }
          if (errors[error]['idNoLengthCheckError']) {
            message = `The ${field} should be of length 9 or 12 characters.`;
          }
          if (errors[error]['noSpecialCharactersError']) {
            message = `The ${field} should not contain special characters.`;
          }
          if (errors[error]['idNoBirthCerError']) {
            message = `The ${field} should be of length 12 characters.`;
          }
        }

        let path = error.split('.').slice(0, -1).join('.');
        if (errorMessages[path]) {
          errorMessages[path].push(message);
        } else {
          errorMessages[path] = [];
          errorMessages[path].push(message);
        }
      }
      for (let validation in errorMessages) {
        let newKey = `${validation}.validations`;
        if (this.modelForm[this.rendererId]?.controls[newKey]) {
          this.modelForm[this.rendererId].controls[newKey].value =
            errorMessages[validation];
        }
      }
      window['NRenderer']?.errorMonitering(
        this.modelForm[this.rendererId].controls,
        errorMessages
      );
    } else {
      if (touchedControls) {
        for (let error in touchedControls) {
          let path = error.split('.').slice(0, -1).join('.');
          let newKey = `${path}.validations`;
          if (this.modelForm[this.rendererId]?.controls[newKey]) {
            this.modelForm[this.rendererId].controls[newKey].value = null;
          }
        }
      }
    }
  }

  isFormGroup(control: AbstractControl): control is FormGroup {
    return !!(<FormGroup>control).controls;
  }

  collectErrors(control: AbstractControl): any | null {
    if (this.isFormGroup(control)) {
      return Object.entries(control.controls).reduce(
        (acc, [key, childControl]) => {
          const childErrors = this.collectErrors(childControl);
          if (childErrors) {
            acc = { ...acc, [key]: childErrors };
          }
          return acc;
        },
        null
      );
    } else {
      return control.errors;
    }
  }

  /**
   * Decides whether the component to be rendered on the UI
   * @param validators - Gets the expression to be evaluted .
   * @returns - A boolean whether to show the component.
   */
  componentValidation(metadata: Metadata) {
    try {
      if (metadata?.expression) {
        return this.nDynamicRendererService.evaluateExpression(
          metadata.expression,
          this.model
        );
      }
      return true;
    } catch (error) {
      this.errorEmit(error);
    }
  }

  /**
   * This function updates the model on every formcontrol change
   */

  parseModel() {
    for (let i in this.modelForm[this.rendererId].value) {
      switch (this.typeMapper[i]) {
        case 'number':
          this.modelForm[this.rendererId].value[i] = Number(
            this.modelForm[this.rendererId].value[i]
          );
          break;
        case 'boolean':
          this.modelForm[this.rendererId].value[i] = Boolean(
            this.modelForm[this.rendererId].value[i]
          );
          break;
        default:
      }
    }
  }

  /**
   * Executes the functions passed as args
   * @param result - Its array of functions that needs to be executed
   */
  async outputMethods(change_event: any) {
    try {
      if (change_event?.functions) {
        await this.nDynamicRendererService.callFunction(
          change_event?.functions,
          this.rendererId,
          this.model,
          this.viewPanels,
          this.schema,
          (updatedModel: any) => {
            this.modelChange.emit(updatedModel);
          },
          (errors: any) => {
            errors?.forEach((error: any) => this.errorEmit(error));
          }
        );
      }
      // if (change_event?.clientNumber && change_event?.section) {
        if(!change_event?.init){
          this.changeDetection.emit(change_event);
        }
      // }
    } catch (error) {
      this.errorEmit(error);
    }
  }

  /**
   * Get the View class and applying the default classes
   * @param stylesClass - Classes from the View.
   * @param category - Category of the styles.
   * @returns - string | string[]  | { [klass: string]: any; }
   */
  getClass(stylesClass: any, category: any) {
    if (stylesClass && typeof stylesClass == 'string') {
      return stylesClass + ' ' + `${category}-default-styles`;
    }
    if (stylesClass && Array.isArray(stylesClass)) {
      stylesClass.push(`${category}-default-styles`);
    }
    if (stylesClass && Object.keys(stylesClass).length > 0) {
      stylesClass[`${category}-default-styles`] == true;
    }
    return `${category}-default-styles`;
  }

  readModel(model_path: any) {
    if (model_path?.startsWith('@')) {
      model_path = model_path.substr(1);
      return this.nDynamicRendererService.readModel(model_path, this.model);
    }
    return model_path;
  }
  /**
   * Validates the object with the schema.
   * @param obj {object} - The object to be validated.
   * @param schema {object} - Schema.
   * @returns {Boolean} - Returns a boolean based on the validation.
   */

  validateObjectWithSchema(obj: any, schema: any, appMetadata?: any) {
    if (appMetadata?.validateSchema) {
      return this.ajv.validate(schema, obj);
    }
    return true;
  }

  /**
   * Error event which fires when ever an error pops up in the component.
   * @param error
   */
  errorEmit(error: any) {
    this.onError.emit(error);
  }
}
